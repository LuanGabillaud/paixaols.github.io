O dataframe da biblioteca pandas possui o método plot que gera histogramas com o parâmetro kind='hist':


import numpy as np
import pandas as pd

# Dados aleatóios
np.random.seed(1)
df = pd.DataFrame({
    'random': np.random.normal(loc=0, scale=1, size=1000)
})

# Histograma
df.plot(
    y='random',
    kind='hist',
    title='Histograma',
    facecolor='green',
    edgecolor='black',
    legend=False,
    figsize=(6,5)
)

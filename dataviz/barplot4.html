Gráfico de barras

Gráfico de barras comparado com dois grupos de dados.


import matplotlib.pyplot as plt
import numpy as np

# Dados
labels = ['A', 'B', 'C', 'D', 'E']
group1 = [2, 5, 3, 5, 7]
group2 = [5, 4, 5, 3, 2]

x = np.arange(len(labels))
width = 0.4

# Criação dos gráficos
rects1 = plt.bar(x - width/2, group1, width, label='Grupo 1')
rects2 = plt.bar(x + width/2, group2, width, label='Grupo 2')

# Textos para auxiliar a visualização
plt.xticks(x, labels)
plt.legend()

# Rótulos com o valor de cada barra
plt.bar_label(rects1, padding=3)
plt.bar_label(rects2, padding=3)
